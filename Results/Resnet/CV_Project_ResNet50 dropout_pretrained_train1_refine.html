def process_result(result) :
    new = []
    for i in range(4):
        if(i<2) :
            temp = []
            for var in result[i]:
                temp.append(var/50)
            new.append(temp)
        else:
            temp = []
            for var in result[i] :
                temp.append(var * 50)
            new.append(temp)
    return new
